  <!-- جہاں ٹائپنگ ٹیکسٹ دکھے گا -->

  <script>
    /* 🔹 وہ جملے جو باری باری لکھے جائیں گے */
    const messages = [
      "🚀 Welcome to the Future 🚀",
      "✨ The best Tpewriter effect 2025 ✨",
      "🌌 Designed by Zohaib Farooq 🌌"
    ];

    let text = "";          // جو لکھا جا رہا ہے
    let index = 0;          // کون سا جملہ ہے
    let isDeleting = false; // کیا مٹا رہے ہیں یا لکھ رہے ہیں؟

    function typeEffect() {
      const subtitleEl = document.getElementById("subtitle");
      const currentMessage = messages[index];

      let speed = 120; // لکھنے کی رفتار
      if (isDeleting) speed = 60; // مٹانے میں تیزی

      /* 🔸 substring() کا استعمال:  
         کسی لفظ یا جملے کے صرف کچھ حروف لینا۔  
         مثال: "Hello".substring(0, 3) → "Hel"
      */

      if (!isDeleting) {
        // ایک ایک کر کے حروف بڑھتے جائیں
        text = currentMessage.substring(0, text.length + 1);
      } else {
        // ایک ایک کر کے حروف مٹتے جائیں
        text = currentMessage.substring(0, text.length - 1);
      }

      subtitleEl.textContent = text;

      /* 💡 setTimeout() کیا ہے؟
         یہ فنکشن کسی کام کو "تھوڑا وقت گزرنے کے بعد" دوبارہ چلاتا ہے۔
         مثال کے طور پر:
         setTimeout(کام, وقت)
         مطلب: اتنے ملی سیکنڈ بعد یہ کام دوبارہ کرنا۔
      */

      // جب پورا جملہ لکھ لیا جائے تو 1 سیکنڈ بعد مٹانا شروع کرو
      if (!isDeleting && text === currentMessage) {
        setTimeout(() => (isDeleting = true), 1000);
      } 
      // جب سب مٹ جائے تو اگلا جملہ شروع کرو
      else if (isDeleting && text === "") {
        isDeleting = false;
        index = (index + 1) % messages.length; // اگلے جملے پر جاؤ
      }

      // فنکشن خود کو دوبارہ بلاتا ہے تاکہ ایفیکٹ چلتا رہے
      setTimeout(typeEffect, speed);
    }

    // صفحہ لوڈ ہوتے ہی ایفیکٹ شروع کرو
    typeEffect();
  </script>

ری ایکٹ JS ٹائپ رائٹر ایفیکٹ (مکمل کوڈ)

import { useState, useEffect } from "react";

function Subtitle() {
  // 🔹 وہ جملے جو باری باری لکھے جائیں گے
  const messages = [
    "🚀 Welcome to the Future 🚀",
    "✨ Powered by React + Vite ✨", 
    "🌌 Designed by Zohaib Farooq. 🌌"
  ];

  // 🔹 ری ایکٹ اسٹیٹس
  const [text, setText] = useState("");          // جو لکھا جا رہا ہے
  const [index, setIndex] = useState(0);         // کون سا جملہ ہے
  const [isDeleting, setIsDeleting] = useState(false); // کیا مٹا رہے ہیں؟

  useEffect(() => {
    const currentMessage = messages[index];
    let speed = 120; // لکھنے کی رفتار
    if (isDeleting) speed = 60; // مٹانے میں تیزی

    const timeout = setTimeout(() => {
      setText(prev =>
        isDeleting
          ? currentMessage.substring(0, prev.length - 1) // ایک حرف مٹاؤ
          : currentMessage.substring(0, prev.length + 1) // ایک حرف بڑھاؤ
      );

      // جب پورا جملہ لکھ لیا جائے تو 1 سیکنڈ بعد مٹانا شروع کرو
      if (!isDeleting && text === currentMessage) {
        setTimeout(() => setIsDeleting(true), 1000);
      } 
      // جب سب مٹ جائے تو اگلا جملہ شروع کرو
      else if (isDeleting && text === "") {
        setIsDeleting(false);
        setIndex((prev) => (prev + 1) % messages.length); // اگلے جملے پر جاؤ
      }
    }, speed);

    // کلین اپ: پرانے ٹائم آؤٹ کو صاف کرو
    return () => clearTimeout(timeout);
  }, [text, isDeleting, index, messages]);

  return <h2 className="subtitle">{text}</h2>;
}

export default Subtitle;

🧠 ری ایکٹ جے ایس میں ٹائپ رائٹر ایفیکٹ - آسان وضاحت

1 Import

import { useState, useEffect } from "react";

مطلب: ہم React کے اندر سے دو خاص "hooks" لے رہے ہیں:

useState — چیزوں کی موجودہ حالت (state) یاد رکھنے کے لیے

useEffect — کوئی کام بار بار یا کسی خاص وقت پر چلانے کے لیے

2 Messages Array

const messages = ["🚀 Welcome to the Future 🚀", "✨ Powered by React + Vite ✨", "🌌 Designed by Zohaib Farooq. 🌌"];

مطلب: یہ تین جملے ہیں جو باری باری اسکرین پر لکھے جائیں گے۔

جیسے: پہلا جملہ لکھا جائے گا، پھر مٹایا جائے گا، پھر دوسرا جملہ آئے گا… اور یہ سلسلہ چلتا رہے گا۔

3 useState Hooks

const [text, setText] = useState("");

const [index, setIndex] = useState(0);

const [isDeleting, setIsDeleting] = useState(false);

مطلب: یہ تین states ہیں:

text → جو لفظ اس وقت اسکرین پر لکھا جا رہا ہے

index → کون سا جملہ چل رہا ہے (پہلا؟ دوسرا؟ تیسرا؟)

isDeleting → کیا ہم لفظ لکھ رہے ہیں یا مٹا رہے ہیں؟

4 useEffect Hook

useEffect(() => { ... }, [text, isDeleting, index, messages]);

مطلب: یہ کوڈ ہر بار چلے گا جب text, isDeleting, یا index بدلے۔

یعنی جب بھی کوئی نیا حرف لکھا یا مٹایا جائے، React دوبارہ ایفیکٹ چلائے گا۔

5 Typing Speed

let speed = 120;

if (isDeleting) speed = 60;

مطلب: جب ہم لکھ رہے ہیں تو رفتار 120ms فی حرف ہے، اور جب مٹا رہے ہیں تو 60ms (تھوڑا تیز) ہے۔

6 Text Change Logic

setText(prev => isDeleting ? currentMessage.substring(0, prev.length - 1) : currentMessage.substring(0, prev.length + 1));

مطلب: اگر ہم لکھ رہے ہیں تو حرف ایک بڑھاؤ، اگر ہم مٹا رہے ہیں تو حرف ایک گھٹاؤ

substring(0, n) ایک لفظ کے پہلے n حروف واپس کرتا ہے۔

مثال: "Hello".substring(0, 3) → نتیجہ "Hel"

7 اگلا مرحلہ کب؟

if (!isDeleting && text === currentMessage) { setTimeout(() => setIsDeleting(true), 1000); }

مطلب: جب پورا جملہ مکمل ہو جائے، تو ایک سیکنڈ بعد اسے مٹانا شروع کرو۔

8 اگلا جملہ کب؟

else if (isDeleting && text === "") { setIsDeleting(false); setIndex((prev) => (prev + 1) % messages.length); }

مطلب: جب مٹانا ختم ہو جائے (یعنی text خالی ہو جائے) تو اگلے جملے پر چلے جاؤ۔

% messages.length اس لیے کہ آخری جملے کے بعد دوبارہ پہلے پر آ جائے۔

9 Cleanup

return () => clearTimeout(timeout);

مطلب: اگر React کمپوننٹ دوبارہ چلے تو پرانے setTimeout کو بند کر دو — ورنہ ایفیکٹ دو بار چلنے لگے گا۔

10 Return JSX

return <h2 className="subtitle">{text}</h2>;

مطلب: React آخر میں وہی چیز دکھائے گا جو text میں لکھی ہوئی ہے۔

یہی آہستہ آستہ بدلتا ہوا جملہ Typewriter کی طرح نظر آتا ہے۔

🧠 سادہ سمجھنے والی وضاحت

🟢 substring(start, end)

یہ String (یعنی لفظ یا جملہ) کا ایک حصہ نکالتا ہے۔

مثلاً:

"Zohaib".substring(0, 3) // نتیجہ: "Zoh"

یہ 0 سے شروع ہوتا ہے اور 3 تک کے حروف لیتا ہے (لیکن 3 والا شامل نہیں ہوتا)۔

ہم نے اسی کو استعمال کیا تاکہ جملہ آہستہ آستہ ایک ایک حرف سے بنتا جائے یا مٹتا جائے۔

🟣 setTimeout(function, time)

یہ فنکشن کمپیوٹر کو کہتا ہے:

"یہ کام اتنے وقت بعد دوبارہ کرنا۔"

مثلاً:

setTimeout(() => console.log("Hi"), 1000);

یہ ایک سیکنڈ بعد "Hi" لکھے گا۔

ہم نے اس کو استعمال کیا تاکہ ٹائپنگ آہستہ آستہ ہوتی رہے اور مسلسل چلتی دکھائی دے۔

🧠 آسان وضاحت (کوڈ کے آخر میں خلاصہ)

messages میں ہم نے وہ تین جملے رکھے ہیں جو باری باری لکھے اور مٹائے جائیں گے۔

text ایک String ہے جو آہستہ آستہ بڑھتی یا گھٹتی ہے — جیسے کوئی بندہ ٹائپ کر رہا ہو۔

isDeleting بتاتا ہے کہ ہم لکھ رہے ہیں یا مٹا رہے ہیں۔

setTimeout(typeEffect, speed) ہر تھوڑی دیر بعد فنکشن دوبارہ چلاتا ہے تاکہ حرکت جاری رہے۔

CSS کا .subtitle::after حصہ کرسر (|) بناتا ہے جو مسلسل جھپکتا رہتا ہے۔

✨ خلاصہ:

messages[] = جملے رکھنے کے لیے

substring() = ایک ایک حرف بڑھانے یا گھٹانے کے لیے

setTimeout() = حرکت بار بار دہرانے کے لیے

CSS ::after + keyframes = جھپکنے والا کرسر دکھانے کے لیے
